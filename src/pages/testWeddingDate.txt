        // Helper function to get Chinese Zodiac
        function getZodiac(year) {
            // Note: A simplified lookup; actual Chinese zodiac changes on the lunar new year
            return zodiacSigns[year] || "Unknown";
        }

        // Helper function to check if two zodiac signs clash
        function doZodiacsClash(zodiac1, zodiac2) {
            return clashPairs[zodiac1] === zodiac2 || clashPairs[zodiac2] === zodiac1;
        }

        // Main logic to find auspicious dates
        function findAuspiciousDates() {
            const brideDate = brideBirthdateEl.value;
            const groomDate = groomBirthdateEl.value;

            if (!brideDate || !groomDate) {
                showModal("Missing Information", "Please enter both the bride's and groom's birthdays to find auspicious dates.");
                return;
            }

            const brideYear = new Date(brideDate).getFullYear();
            const groomYear = new Date(groomDate).getFullYear();
            const brideZodiac = getZodiac(brideYear);
            const groomZodiac = getZodiac(groomYear);

            const auspiciousDates = [];
            const today = new Date();
            const endDate = new Date();
            endDate.setMonth(today.getMonth() + 6); // Look for dates in the next 6 months

            for (let d = new Date(today); d <= endDate; d.setDate(d.getDate() + 1)) {
                const day = d.getDate();
                const month = d.getMonth() + 1;
                const year = d.getFullYear();
                const dayOfWeek = d.getDay(); // 0 = Sunday, 1 = Monday, etc.

                let isAuspicious = true;
                const reasons = [];

                // Rule 1: Avoid zodiac clashes on the day's zodiac (Simplified)
                // This is a simplified check for demo purposes. Real Feng Shui uses complex calculations.
                const dayZodiac = getZodiac(year);
                if (doZodiacsClash(brideZodiac, dayZodiac) || doZodiacsClash(groomZodiac, dayZodiac)) {
                    isAuspicious = false;
                } else {
                    reasons.push(`<i class="fas fa-check-circle text-green-500 mr-2"></i> No Zodiac Clash`);
                }

                // Rule 2: Avoid unlucky numbers (e.g., 4)
                if (day.toString().includes('4')) {
                    isAuspicious = false;
                }

                // Rule 3: Favorable numbers (e.g., 2, 8, 9)
                if (day > 0 && luckyNumbers.has(day % 10)) {
                    reasons.push(`<i class="fas fa-star text-yellow-500 mr-2"></i> Lucky Day Number (${day})`);
                }

                // Rule 4: Even months for harmony
                if (month % 2 !== 0) {
                    // This is a simplified rule. In Chinese tradition, months are lunar.
                    // For Gregorian calendar, this is a fun approximation.
                    isAuspicious = false;
                } else {
                    reasons.push(`<i class="fas fa-balance-scale text-blue-500 mr-2"></i> Even Month for Harmony`);
                }

                // Rule 5: Weekend dates are practical and often considered good
                if (dayOfWeek === 6 || dayOfWeek === 0) {
                     reasons.push(`<i class="fas fa-calendar-check text-purple-500 mr-2"></i> Favorable Weekend`);
                }

                // Add more complex checks here if a real-time lunar calendar calculation was available
                // E.g., Avoid "Ghost Month" (7th Lunar Month) and "San Niang Sha" days.

                if (isAuspicious && reasons.length > 0) {
                    auspiciousDates.push({ date: new Date(d), reasons: reasons });
                }
            }

            displayResults(auspiciousDates);
        }

                const showPaywall = dates.length > 3;

                        <div class="space-y-2 text-sm text-gray-700">

                        </div>
                    `;
                    dateResultsEl.appendChild(card);
                });

                if (showPaywall) {
